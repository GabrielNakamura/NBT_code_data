regions_1750 <- flow_period_region2 |> 
  dplyr::filter(period == 1750) |> 
  dplyr::select(-c(period, total_period_region_type)) |> 
  dplyr::mutate(
    region_type = glue::glue(" {region_type}"),
    region_museum = glue::glue("{region_museum} ")) |> 
  dplyr::rename(
    from = region_type,
    to = region_museum,
    value = n) |>
